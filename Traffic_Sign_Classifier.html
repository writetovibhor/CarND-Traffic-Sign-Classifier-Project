    r1 = np.float32([[0,0],[image.shape[1],0],[image.shape[1],image.shape[0]],[0,image.shape[0]]])
    r2 = np.float32([[0,0],[image.shape[0] - OFFSET,OFFSET],[image.shape[1]-OFFSET,image.shape[0]-OFFSET],[0,image.shape[0]]])
    l1 = np.float32([[0,0],[image.shape[1],0],[image.shape[1],image.shape[0]],[0,image.shape[0]]])
    l2 = np.float32([[OFFSET,OFFSET],[image.shape[1],0],[image.shape[1],image.shape[0]],[OFFSET,image.shape[0]-OFFSET]])
    Mr = cv2.getPerspectiveTransform(r1,r2)
    Ml = cv2.getPerspectiveTransform(l1,l2)

    def changePerspective(frame, edge):
        res = cv2.resize(frame,None,fx=scale, fy=scale, interpolation = cv2.INTER_CUBIC)
        if edge == 'right':
            changed = cv2.warpPerspective(res,Mr,(res.shape[1], res.shape[0]))
            y_offset = int((changed.shape[0] - frame.shape[0]) * 0.5)
            return changed[y_offset:y_offset + frame.shape[0], 0:frame.shape[1]]
        elif edge == 'left':
            changed = cv2.warpPerspective(res,Ml,(res.shape[1],res.shape[0]))
            y_offset = int((changed.shape[0] - frame.shape[0]) * 0.5)
            return changed[y_offset:y_offset + frame.shape[0], changed.shape[1] - frame.shape[1]:changed.shape[1]]

    def equalize(frame):
        yuv = cv2.cvtColor(frame, cv2.COLOR_RGB2YUV)
        yuv[:,:,0] = cv2.equalizeHist(yuv[:,:,0])
        return cv2.cvtColor(yuv, cv2.COLOR_YUV2RGB)

    def grayscale(frame):
        gray = cv2.cvtColor(frame, cv2.COLOR_RGB2GRAY)

        return gray3ch

from tensorflow.contrib.layers import flatten
from tensorflow.contrib.layers import xavier_initializer
def LeNet(x):    
    class Specs:
        class Conv1:
            Filter = [5, 5, 3, 18]
            Stride = [1, 1, 1, 1]
            Padding = 'VALID'
            Bias = Filter[3]
        class Pool1:
            Kernel_Size = [1, 2, 2, 1]
            Stride = [1, 2, 2, 1]
            Padding = 'VALID'
        class Conv2:
            Filter = [5, 5, 18, 48]
            Stride = [1, 1, 1, 1]
            Padding = 'VALID'
            Bias = Filter[3]
        class Pool2:
            Kernel_Size = [1, 2, 2, 1]
            Stride = [1, 2, 2, 1]
            Padding = 'VALID'
        class Dense1:
            Units = 360
        class Dense2:
            Units = 252
        class Dense3:
            Units = n_classes

    # TODO: Layer 1: Convolutional. Input = 32x32x3. Output = 28x28x18.
    #filter1 = tf.Variable(tf.truncated_normal(Specs.Conv1.Filter, mean=0, stddev=0.1), name='filter1')
    filter1 = tf.get_variable(name='filter1', shape=Specs.Conv1.Filter, initializer=xavier_initializer())
    bias1 = tf.Variable(tf.zeros(Specs.Conv1.Bias), name='bias1')
    conv1 = tf.nn.conv2d(x, filter1, Specs.Conv1.Stride, Specs.Conv1.Padding, name='conv1') + bias1
    norm1 = tf.nn.l2_normalize(conv1, 2, name='norm1')
    
    # TODO: Activation.
    relu1 = tf.nn.relu(norm1, name='relu1')

    # TODO: Pooling. Input = 28x28x18. Output = 14x14x18.
    pool1 = tf.nn.max_pool(relu1, Specs.Pool1.Kernel_Size, Specs.Pool1.Stride, Specs.Pool1.Padding, name='pool1')

    # TODO: Layer 2: Convolutional. Output = 10x10x48.
    #filter2 = tf.Variable(tf.truncated_normal(Specs.Conv2.Filter, mean=0, stddev=0.1), name='filter2')
    filter2 = tf.get_variable(name='filter2', shape=Specs.Conv2.Filter, initializer=xavier_initializer())
    bias2 = tf.Variable(tf.zeros(Specs.Conv2.Bias), name='bias2')
    conv2 = tf.nn.conv2d(pool1, filter2, Specs.Conv2.Stride, Specs.Conv2.Padding, name='conv2') + bias2
    norm2 = tf.nn.l2_normalize(conv2, 2, name='norm2')
    
    # TODO: Activation.
    relu2 = tf.nn.relu(norm2, name='relu2')

    # TODO: Pooling. Input = 10x10x48. Output = 5x5x48.
    pool2 = tf.nn.max_pool(relu2, Specs.Pool2.Kernel_Size, Specs.Pool2.Stride, Specs.Pool2.Padding, name='pool2')

    # TODO: Flatten. Input = 5x5x48. Output = 1200.
    flat1 = flatten(pool2)
    
    # TODO: Layer 3: Fully Connected. Input = 1200. Output = 360.
    dense1 = tf.layers.dense(inputs=flat1, units=Specs.Dense1.Units, name='dense1')
    
    # TODO: Activation.
    relu3 = tf.nn.relu(dense1, name='relu3')
    
    # Dropout
    dropout1 = tf.nn.dropout(relu3, keep_prob, name='dropout1')

    # TODO: Layer 4: Fully Connected. Input = 360. Output = 252.
    dense2 = tf.layers.dense(inputs=dropout1, units=Specs.Dense2.Units, name='dense2')

    # TODO: Activation.
    relu4 = tf.nn.relu(dense2, name='relu4')

    # TODO: Layer 5: Fully Connected. Input = 252. Output = 43.
    logits = tf.layers.dense(inputs=relu4, units=Specs.Dense3.Units, name='dense3')
    
    return logits

correct_prediction = tf.equal(tf.argmax(logits, 1), tf.argmax(one_hot_y, 1))
accuracy_operation = tf.reduce_mean(tf.cast(correct_prediction, tf.float32))
saver = tf.train.Saver()

def evaluate(X_data, y_data):
    num_examples = len(X_data)
    total_accuracy = 0
    sess = tf.get_default_session()
    for offset in range(0, num_examples, BATCH_SIZE):
        batch_x, batch_y = X_data[offset:offset+BATCH_SIZE], y_data[offset:offset+BATCH_SIZE]
        accuracy = sess.run(accuracy_operation, feed_dict={x: batch_x, y: batch_y, keep_prob: 1.0})
        total_accuracy += (accuracy * len(batch_x))
    return total_accuracy / num_examples

Step 4: Visualize the Neural Network's State with Test Images¶

This Section is not required to complete but acts as an additional excersise for understaning the output of a neural network's weights. While neural networks can be a great learning device they are often referred to as a black box. We can understand what the weights of a neural network look like better by plotting their feature maps. After successfully training your neural network you can see what it's feature maps look like by plotting the output of the network's weight layers in response to a test stimuli image. From these plotted feature maps, it's possible to see what characteristics of an image the network finds interesting. For a sign, maybe the inner network feature maps react with high activation to the sign's boundary outline or to the contrast in the sign's painted symbol.

Provided for you below is the function code that allows you to get the visualization output of any tensorflow weight layer you want. The inputs to the function should be a stimuli image, one used during training or a new one you provided, and then the tensorflow variable name that represents the layer's state during the training process, for instance if you wanted to see what the LeNet lab's feature maps looked like for it's second convolutional layer you could enter conv2 as the tf_activation variable.

For an example of what feature map outputs look like, check out NVIDIA's results in their paper End-to-End Deep Learning for Self-Driving Cars in the section Visualization of internal CNN State. NVIDIA was able to show that their network's inner weights had high activations to road boundary lines by comparing feature maps from an image with a clear path to one without. Try experimenting with a similar test to show that your trained network's weights are looking for interesting features, whether it's looking at differences in feature maps from images with or without a sign, or even what feature maps look like in a trained network vs a completely untrained one on the same sign image.
